\newcommand{\roundphoto}[2][]{%
  \begin{tikzpicture}[#1]
    \node[circle, minimum size=2.6cm, line width=2.5pt,
          draw=accent!85!white, inner sep=0pt,
          path picture={
            \node at (path picture bounding box.center){
              \includegraphics[width=2.5cm,height=2.5cm,keepaspectratio]{#2}};
          },
          blur shadow={shadow blur steps=8, shadow xshift=0, shadow yshift=2pt}]{};
  \end{tikzpicture}%
}
